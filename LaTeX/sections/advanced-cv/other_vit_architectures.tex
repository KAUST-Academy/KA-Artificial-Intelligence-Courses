\begin{frame}{Object Detection with Transformers: DETR}
\only<1>{
\begin{itemize}
    \item Simple object detection pipeline: directly outputs a set of boxes from a Transformer
    \item No anchors, no regression of box transforms
    \item Matches predicted boxes to ground truth boxes with bipartite matching; trains to regress box coordinates
\end{itemize}
\footnotetext{Carion et al., ``End-to-End Object Detection with Transformers,'' ECCV 2020}
